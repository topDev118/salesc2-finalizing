select DISTINCT team_manager.id userid, team_manager.name username, team_manager.lastactivity lastactivity, id avatar, team_manager.id link, cometchat_status.message, cometchat_status.status from team_manager left join cometchat_status on team_manager.id = cometchat_status.userid   where team_manager.id <> '3' and (cometchat_status.status IS NULL OR cometchat_status.status <> 'invisible' OR cometchat_status.status <> 'offline') order by username asc
